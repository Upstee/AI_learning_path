```python
# 测试1：Python基础
def test_python_basics():
    """测试Python基础能力"""
    # 1. 变量和数据类型
    name = "AI学习者"
    age = 20
    scores = [85, 90, 88, 92]
    
    # 2. 控制流
    if age >= 18:
        print(f"{name}已成年")
    
    # 3. 循环
    total = 0
    for score in scores:
        total += score
    average = total / len(scores)
    print(f"平均分：{average}")
    
    # 4. 函数
    def calculate_grade(score):
        if score >= 90:
            return "A"
        elif score >= 80:
            return "B"
        else:
            return "C"
    
    return calculate_grade(average)

# 测试2：数据结构
def test_data_structures():
    """测试数据结构使用"""
    # 列表
    numbers = [1, 2, 3, 4, 5]
    squared = [x**2 for x in numbers]
    
    # 字典
    student = {
        "name": "张三",
        "age": 20,
        "scores": [85, 90, 88]
    }
    
    # 集合
    unique_numbers = set([1, 2, 2, 3, 3, 4])
    
    return squared, student, unique_numbers

# 测试3：文件操作
def test_file_operations():
    """测试文件操作"""
    # 写入文件
    with open("test.txt", "w", encoding="utf-8") as f:
        f.write("Hello, AI!")
    
    # 读取文件
    with open("test.txt", "r", encoding="utf-8") as f:
        content = f.read()
    
    return content

# 运行测试
if __name__ == "__main__":
    print("=== Python基础测试 ===")
    grade = test_python_basics()
    print(f"等级：{grade}")
    
    squared, student, unique = test_data_structures()
    print(f"平方数：{squared}")
    print(f"学生信息：{student}")
    print(f"唯一数字：{unique}")
    
    content = test_file_operations()
    print(f"文件内容：{content}")
```

### 4.2 数学能力测试

```python
import numpy as np

# 测试1：线性代数
def test_linear_algebra():
    """测试线性代数能力"""
    # 创建矩阵
    A = np.array([[1, 2], [3, 4]])
    B = np.array([[5, 6], [7, 8]])
    
    # 矩阵运算
    C = A + B  # 矩阵加法
    D = A @ B  # 矩阵乘法
    E = A.T   # 转置
    
    # 特征值
    eigenvalues, eigenvectors = np.linalg.eig(A)
    
    return C, D, E, eigenvalues

# 测试2：概率统计
def test_probability():
    """测试概率统计能力"""
    # 生成随机数
    data = np.random.normal(0, 1, 1000)
    
    # 统计量
    mean = np.mean(data)
    std = np.std(data)
    
    # 概率分布
    from scipy import stats
    normal_dist = stats.norm(0, 1)
    prob = normal_dist.pdf(0)  # 概率密度
    
    return mean, std, prob

# 测试3：微积分
def test_calculus():
    """测试微积分能力"""
    # 数值求导
    def f(x):
        return x**2
    
    # 使用数值方法求导
    x = 2.0
    h = 0.0001
    derivative = (f(x + h) - f(x)) / h
    
    return derivative

# 运行测试
if __name__ == "__main__":
    print("=== 数学能力测试 ===")
    C, D, E, eigenvals = test_linear_algebra()
    print(f"矩阵加法结果：\n{C}")
    print(f"特征值：{eigenvals}")
    
    mean, std, prob = test_probability()
    print(f"均值：{mean:.4f}, 标准差：{std:.4f}")
    
    derivative = test_calculus()
    print(f"x=2时，x^2的导数：{derivative:.4f}（理论值：4.0）")
```
