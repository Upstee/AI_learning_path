### 4.1 学习进度跟踪脚本

```python
# 学习进度跟踪系统
import json
from datetime import datetime
from pathlib import Path

class LearningTracker:
    def __init__(self, plan_file="learning_plan.json"):
        self.plan_file = plan_file
        self.plan = self.load_plan()
    
    def load_plan(self):
        """加载学习计划"""
        if Path(self.plan_file).exists():
            with open(self.plan_file, 'r', encoding='utf-8') as f:
                return json.load(f)
        else:
            return self.create_default_plan()
    
    def create_default_plan(self):
        """创建默认学习计划"""
        return {
            "stages": [
                {"name": "阶段0：准备阶段", "duration": "1-2个月", "status": "未开始"},
                {"name": "阶段1：Python进阶", "duration": "2-3个月", "status": "未开始"},
                {"name": "阶段2：数学基础", "duration": "2-3个月", "status": "未开始"},
                {"name": "阶段3：数据处理基础", "duration": "1-2个月", "status": "未开始"},
                {"name": "阶段4：机器学习基础", "duration": "3-4个月", "status": "未开始"},
                {"name": "阶段5：深度学习基础", "duration": "3-4个月", "status": "未开始"},
                {"name": "阶段6：专业方向选择", "duration": "6-12个月", "status": "未开始"},
                {"name": "阶段7：高级专题", "duration": "6-12个月", "status": "未开始"},
                {"name": "阶段8：综合项目", "duration": "持续", "status": "未开始"}
            ],
            "start_date": datetime.now().strftime("%Y-%m-%d"),
            "current_stage": 0,
            "notes": []
        }
    
    def update_stage(self, stage_index, status):
        """更新阶段状态"""
        if 0 <= stage_index < len(self.plan["stages"]):
            self.plan["stages"][stage_index]["status"] = status
            self.save_plan()
            print(f"已更新：{self.plan['stages'][stage_index]['name']} -> {status}")
        else:
            print("无效的阶段索引")
    
    def add_note(self, note):
        """添加学习笔记"""
        self.plan["notes"].append({
            "date": datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            "content": note
        })
        self.save_plan()
    
    def save_plan(self):
        """保存学习计划"""
        with open(self.plan_file, 'w', encoding='utf-8') as f:
            json.dump(self.plan, f, ensure_ascii=False, indent=2)
    
    def show_progress(self):
        """显示学习进度"""
        print("\n=== 学习进度 ===")
        total = len(self.plan["stages"])
        completed = sum(1 for s in self.plan["stages"] if s["status"] == "已完成")
        in_progress = sum(1 for s in self.plan["stages"] if s["status"] == "进行中")
        
        print(f"总阶段数：{total}")
        print(f"已完成：{completed}")
        print(f"进行中：{in_progress}")
        print(f"未开始：{total - completed - in_progress}")
        print(f"完成率：{completed/total*100:.1f}%")
        
        print("\n=== 各阶段状态 ===")
        for i, stage in enumerate(self.plan["stages"]):
            status_icon = "✓" if stage["status"] == "已完成" else "→" if stage["status"] == "进行中" else "○"
            print(f"{status_icon} [{i}] {stage['name']} ({stage['duration']}) - {stage['status']}")

# 使用示例
if __name__ == "__main__":
```
